33 // The idea is that infinity - infinity gives NaN, and NaN != NaN is true.
34 //
35 // This should work on any IEEE-754-compliant implementation, which is
36 // another way of saying that it does not always work (in particular,
37 // according to stackoverflow, it won't work with gcc with the --fast-math
38 // option), but I know of no better way of testing this portably in C++ prior
39 // to C++11. When soci moves to C++11 this should be replaced
40 // with std::isfinite().
41 template <typename T>
42 bool is_infinity_or_nan(T x)
43 {
44  T y = x - x;
45 
46  // We really need exact floating point comparison here.
47  GCC_WARNING_SUPPRESS(float-equal)
48 
49  return (y != y);
50 
51  GCC_WARNING_RESTORE(float-equal)
52 }
