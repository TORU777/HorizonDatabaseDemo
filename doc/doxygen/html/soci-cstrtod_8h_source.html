22 // Locale-independent, i.e. always using "C" locale, function for converting
23 // strings to numbers.
24 //
25 // The string must contain a floating point number in "C" locale, i.e. using
26 // point as decimal separator, and nothing but it. If it does, the converted
27 // number is returned, otherwise an exception is thrown.
28 inline
29 double cstring_to_double(char const* s)
30 {
31  // Unfortunately there is no clean way to parse a number in C locale
32  // without this hack: normally, using std::istringstream with classic
33  // locale should work, but some standard library implementations are buggy
34  // and handle non-default locale in thread-unsafe way, by changing the
35  // global C locale which is unacceptable as it introduces subtle bugs in
36  // multi-thread programs. So we rely on just the standard C functions and
37  // try to make them work by tweaking the input into the form appropriate
38  // for the current locale.
39 
40  // First try with the original input.
41  char* end;
42  double d = strtod(s, &end);
43 
44  bool parsedOK;
45  if (*end == '.')
46  {
47  // Parsing may have stopped because the current locale uses something
48  // different from the point as decimal separator, retry with a comma.
49  //
50  // In principle, values other than point or comma are possible but they
51  // don't seem to be used in practice, so for now keep things simple.
52  size_t const bufSize = strlen(s) + 1;
53  char* const buf = new char[bufSize];
54  strcpy(buf, s);
55  buf[end - s] = ',';
56  d = strtod(buf, &end);
57  parsedOK = end != buf && *end == '\0';
58  delete [] buf;
59  }
60  else
61  {
62  // Notice that we must detect false positives as well: parsing a string
63  // using decimal comma should fail when using this function.
64  parsedOK = end != s && *end == '\0' && !strchr(s, ',');
65  }
66 
67  if (!parsedOK)
68  {
69  throw soci_error(std::string("Cannot convert data: string \"") + s + "\" "
70  "is not a number.");
71  }
72 
73  return d;
74 }
