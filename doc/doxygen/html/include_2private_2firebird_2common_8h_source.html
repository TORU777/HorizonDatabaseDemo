42 template <typename IntType>
43 const char *str2dec(const char * s, IntType &out, short &scale)
44 {
45  int sign = 1;
46  if ('+' == *s)
47  ++s;
48  else if ('-' == *s)
49  {
50  sign = -1;
51  ++s;
52  }
53  scale = 0;
54  bool period = false;
55  IntType res = 0;
56  for (out = 0; *s; ++s, out = res)
57  {
58  if (*s == '.')
59  {
60  if (period)
61  return s;
62  period = true;
63  continue;
64  }
65  int d = *s - '0';
66  if (d < 0 || d > 9)
67  return s;
68  res = res * 10 + static_cast<IntType>(d * sign);
69  if (1 == sign)
70  {
71  if (res < out)
72  return s;
73  }
74  else
75  {
76  if (res > out)
77  return s;
78  }
79  if (period)
80  ++scale;
81  }
82  return s;
83 }

115 template<typename T1>
116 void to_isc(void * val, XSQLVAR * var, short x_scale = 0)
117 {
118  T1 value = *reinterpret_cast<T1*>(val);
119  short scale = var->sqlscale + x_scale;
120  short type = var->sqltype & ~1;
121  long long divisor = 1, multiplier = 1;
122 
123  cond_to_isc<std::numeric_limits<T1>::is_integer>::checkInteger(scale,type);
124 
125  for (int i = 0; i > scale; --i)
126  multiplier *= 10;
127  for (int i = 0; i < scale; ++i)
128  divisor *= 10;
129 
130  switch (type)
131  {
132  case SQL_SHORT:
133  {
134  short tmp = static_cast<short>(round_for_isc(value*multiplier)/divisor);
135  std::memcpy(var->sqldata, &tmp, sizeof(short));
136  }
137  break;
138  case SQL_LONG:
139  {
140  int tmp = static_cast<int>(round_for_isc(value*multiplier)/divisor);
141  std::memcpy(var->sqldata, &tmp, sizeof(int));
142  }
143  break;
144  case SQL_INT64:
145  {
146  long long tmp = static_cast<long long>(round_for_isc(value*multiplier)/divisor);
147  std::memcpy(var->sqldata, &tmp, sizeof(long long));
148  }
149  break;
150  case SQL_FLOAT:
151  {
152  float sql_value = static_cast<float>(value);
153  std::memcpy(var->sqldata, &sql_value, sizeof(float));
154  }
155  break;
156  case SQL_DOUBLE:
157  {
158  double sql_value = static_cast<double>(value);
159  std::memcpy(var->sqldata, &sql_value, sizeof(double));
160  }
161  break;
162  default:
163  throw soci_error("Incorrect data type for numeric conversion");
164  }
165 }

167 template<typename IntType, typename UIntType>
168 void parse_decimal(void * val, XSQLVAR * var, const char * s)
169 {
170  short scale;
171  UIntType t1;
172  IntType t2;
173  if (!*str2dec(s, t1, scale))
174  std::memcpy(val, &t1, sizeof(t1));
175  else if (!*str2dec(s, t2, scale))

177  else
178  throw soci_error("Could not parse decimal value.");
179  to_isc<IntType>(val, var, scale);

182 template<typename IntType>
183 std::string format_decimal(const void *sqldata, int sqlscale)
184 {
185  IntType x = *reinterpret_cast<const IntType *>(sqldata);
186  std::stringstream out;
187  out << x;
188  std::string r = out.str();
189  if (sqlscale < 0)
190  {
191  if (static_cast<int>(r.size()) - (x < 0) <= -sqlscale)
192  {
193  r = std::string(size_t(x < 0), '-') +
194  std::string(-sqlscale - (r.size() - (x < 0)) + 1, '0') +
195  r.substr(size_t(x < 0), std::string::npos);
196  }
197  return r.substr(0, r.size() + sqlscale) + '.' +
198  r.substr(r.size() + sqlscale, std::string::npos);
199  }
200  return r + std::string(sqlscale, '0');
201 }

219 template<typename T1>
220 T1 from_isc(XSQLVAR * var)
221 {
222  short scale = var->sqlscale;
223  T1 tens = 1;
224 
225  if (scale < 0)
226  {
227  cond_from_isc<std::numeric_limits<T1>::is_integer>::checkInteger(scale);
228  for (int i = 0; i > scale; --i)
229  {
230  tens *= 10;
231  }
232  }
233 
234  switch (var->sqltype & ~1)
235  {
236  case SQL_SHORT:
237  return static_cast<T1>(*reinterpret_cast<short*>(var->sqldata)/tens);
238  case SQL_LONG:
239  return static_cast<T1>(*reinterpret_cast<int*>(var->sqldata)/tens);
240  case SQL_INT64:
241  return static_cast<T1>(*reinterpret_cast<long long*>(var->sqldata)/tens);
242  case SQL_FLOAT:
243  return static_cast<T1>(*reinterpret_cast<float*>(var->sqldata));
244  case SQL_DOUBLE:
245  return static_cast<T1>(*reinterpret_cast<double*>(var->sqldata));
246  default:
247  throw soci_error("Incorrect data type for numeric conversion");
