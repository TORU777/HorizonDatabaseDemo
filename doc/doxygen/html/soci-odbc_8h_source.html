332 class SOCI_ODBC_DECL odbc_soci_error : public soci_error
333 {
334  SQLCHAR message_[SQL_MAX_MESSAGE_LENGTH + 1];
335  SQLCHAR sqlstate_[SQL_SQLSTATE_SIZE + 1];
336  SQLINTEGER sqlcode_;
337 
338 public:
339  odbc_soci_error(SQLSMALLINT htype,
340  SQLHANDLE hndl,
341  std::string const & msg)
342  : soci_error(interpret_odbc_error(htype, hndl, msg))
343  {
344  }
345 
346  SQLCHAR const * odbc_error_code() const
347  {
348  return sqlstate_;
349  }
350  SQLINTEGER native_error_code() const
351  {
352  return sqlcode_;
353  }
354  SQLCHAR const * odbc_error_message() const
355  {
356  return message_;
357  }
358 private:
359  std::string interpret_odbc_error(SQLSMALLINT htype, SQLHANDLE hndl, std::string const& msg)
360  {
361  const char* socierror = NULL;
362 
363  SQLSMALLINT length, i = 1;
364  switch ( SQLGetDiagRecA(htype, hndl, i, sqlstate_, &sqlcode_,
365  message_, SQL_MAX_MESSAGE_LENGTH + 1,
366  &length) )
367  {
368  case SQL_SUCCESS:
369  // The error message was successfully retrieved.
370  break;
371 
372  case SQL_INVALID_HANDLE:
373  socierror = "[SOCI]: Invalid handle.";
374  break;
375 
376  case SQL_ERROR:
377  socierror = "[SOCI]: SQLGetDiagRec() error.";
378  break;
379 
380  case SQL_SUCCESS_WITH_INFO:
381  socierror = "[SOCI]: Error message too long.";
382  break;
383 
384  case SQL_NO_DATA:
385  socierror = "[SOCI]: No error.";
386  break;
387 
388  default:
389  socierror = "[SOCI]: Unexpected SQLGetDiagRec() return value.";
390  break;
391  }
392 
393  if (socierror)
394  {
395  // Use our own error message if we failed to retrieve the ODBC one.
396  strcpy(reinterpret_cast<char*>(message_), socierror);
397 
398  // Use "General warning" SQLSTATE code.
399  strcpy(reinterpret_cast<char*>(sqlstate_), "01000");
400 
401  sqlcode_ = 0;
402  }
403 
404  std::ostringstream ss;
405  ss << "Error " << msg << ": " << message_ << " (SQL state " << sqlstate_ << ")";
406 
407  return ss.str();
408  }
409 };

423 inline bool odbc_standard_type_backend_base::use_string_for_bigint() const
424 {
425  // Oracle ODBC driver doesn't support SQL_C_[SU]BIGINT data types
426  // (see appendix G.1 of Oracle Database Administrator's reference at
427  // http://docs.oracle.com/cd/B19306_01/server.102/b15658/app_odbc.htm),
428  // so we need a special workaround for this case and we represent 64
429  // bit integers as strings and rely on ODBC driver for transforming
430  // them to SQL_NUMERIC.
431  return statement_.session_.get_database_product()
432  == odbc_session_backend::prod_oracle;
433 }
