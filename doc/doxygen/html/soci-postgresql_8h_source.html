48 // A class thinly encapsulating PGresult. Its main purpose is to ensure that
49 // PQclear() is always called, avoiding result memory leaks.
50 class postgresql_result
51 {
52 public:
53  // Creates a wrapper for the given, possibly NULL, result. The wrapper
54  // object takes ownership of the object and will call PQclear() on it.
55  explicit postgresql_result(PGresult* result = NULL)
56  {
57  init(result);
58  }
59 
60  // Frees any currently stored result pointer and takes ownership of the
61  // given one.
62  void reset(PGresult* result = NULL)
63  {
64  free();
65  init(result);
66  }
67 
68  // Check whether the status is PGRES_COMMAND_OK and throw an exception if
69  // it is different. Notice that if the query can return any results,
70  // check_for_data() below should be used instead to verify whether anything
71  // was returned or not.
72  //
73  // The provided error message is used only for the exception being thrown
74  // and should describe the operation which yielded this result.
75  void check_for_errors(char const* errMsg) const;
76 
77  // Check whether the status indicates successful query completion, either
78  // with the return results (in which case true is returned) or without them
79  // (then false is returned). If the status corresponds to an error, throws
80  // an exception, just as check_for_errors().
81  bool check_for_data(char const* errMsg) const;
82 
83  // Implicit conversion to const PGresult: this is somewhat dangerous but
84  // allows us to avoid changing the existing code that uses PGresult and
85  // avoids the really bad problem with calling PQclear() twice accidentally
86  // as this would require a conversion to non-const pointer that we do not
87  // provide.
88  operator const PGresult*() const { return result_; }
89 
90  // Get the associated result (which may be NULL). Unlike the implicit
91  // conversion above, this one returns a non-const pointer, so you should be
92  // careful to avoid really modifying it.
93  PGresult* get_result() const { return result_; }
94 
95  // Dtor frees the result.
96  ~postgresql_result() { free(); }
97 
98 private:
99  void init(PGresult* result)
100  {
101  result_ = result;
102  }
103 
104  void free()
105  {
106  // Notice that it is safe to call PQclear() with NULL pointer, it
107  // simply does nothing in this case.
108  PQclear(result_);
109  }
110 
111  PGresult* result_;
112 
113  SOCI_NOT_COPYABLE(postgresql_result)
114 };
